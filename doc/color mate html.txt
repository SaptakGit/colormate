// ── CALCULATOR ──
var rates = { interior: 12, exterior: 18, texture: 35, waterproofing: 45 };
var qMult = { standard: 1, premium: 1.42, luxury: 1.88 };
var cAdd = { good: 0, minor: 1600, major: 3800 };
function calcUpdate() {
  var area = parseInt(document.getElementById('areaSlider').value);
  var coats = parseInt(document.getElementById('coatsSlider').value);
  var stype = document.querySelector('input[name="stype"]:checked').value;
  var qual = document.querySelector('input[name="qual"]:checked').value;
  var cond = document.querySelector('input[name="cond"]:checked').value;
  document.getElementById('areaVal').textContent = area.toLocaleString('en-IN') + ' sq ft';
  document.getElementById('coatsVal').textContent = coats + (coats === 1 ? ' coat' : ' coats');
  var rate = rates[stype] * qMult[qual];
  var mat = Math.round(area * rate * 0.55 * (1 + (coats - 1) * 0.28));
  var labour = Math.round(area * rate * 0.35 * (1 + (coats - 1) * 0.18));
  var primer = Math.round(area * (stype === 'waterproofing' ? 0.8 : 1.2) + cAdd[cond]);
  var low = mat + labour + primer;
  var high = Math.round(low * 1.19);
  var days = Math.max(1, Math.round(area / 420 * (coats * 0.72)));
  document.getElementById('calcTotal').textContent = 'Rs' + low.toLocaleString('en-IN');
  document.getElementById('calcRange').textContent = '– Rs' + high.toLocaleString('en-IN') + ' (depending on brand)';
  document.getElementById('calcMat').textContent = 'Rs' + mat.toLocaleString('en-IN');
  document.getElementById('calcLabour').textContent = 'Rs' + labour.toLocaleString('en-IN');
  document.getElementById('calcPrimer').textContent = 'Rs' + primer.toLocaleString('en-IN');
  document.getElementById('calcDays').textContent = days + '–' + (days + 1) + ' days';
}
calcUpdate();

  var svcHtml = a.services.map(function(s) {
    var icons = { Interior: '🎨', Exterior: '🏠', Waterproofing: '💧', Texture: '✨' };
    var icon = '🖌️';
    Object.keys(icons).forEach(function(k) { if (s.indexOf(k) > -1) icon = icons[k]; });
    return '<div class="ar-sp" onclick="goPage(\'services\')">' + icon + ' ' + s + '</div>';

  var allBtn = document.createElement('button');
  allBtn.className = 'bl-cb' + (cat === 'all' ? ' active' : '');
  allBtn.textContent = 'All Posts';
  allBtn.onclick = function() { renderBlog('all', ''); };
  catBar.appendChild(allBtn);
  cats.forEach(function(c) {
    var b = document.createElement('button');
    b.className = 'bl-cb' + (cat === c ? ' active' : '');
    b.textContent = c;
    b.onclick = function() { renderBlog(c, ''); };
    catBar.appendChild(b);
  });
  var posts = blogPosts.filter(function(p) {
    var mc = cat === 'all' || p.cat === cat;
    var ms = !search || p.title.toLowerCase().indexOf(search.toLowerCase()) > -1 || p.excerpt.toLowerCase().indexOf(search.toLowerCase()) > -1;
    return mc && ms;
  });
  var featEl = document.getElementById('blFeat');
  var feat = (cat === 'all' && !search) ? posts.find(function(p) { return p.featured; }) : null;
  if (feat) {
    featEl.style.display = 'grid';
    featEl.innerHTML = '<div class="bl-fi" style="background:' + feat.bg + '"><span class="bl-fbadge">Featured</span>' + feat.emoji + '</div><div class="bl-fb"><div class="bl-fc">' + feat.cat + '</div><h2 class="bl-ft">' + feat.title + '</h2><p class="bl-fe">' + feat.excerpt + '</p><div class="bl-fm"><span>by ' + feat.author + '</span><span>' + feat.date + '</span><span>' + feat.rt + ' read</span></div><button class="bl-rb" onclick="openArticle(\'' + feat.id + '\')">Read article &rarr;</button></div>';
    posts = posts.filter(function(p) { return p.id !== feat.id; });
  } else {
    featEl.style.display = 'none';
    featEl.innerHTML = '';
  }
  var gridEl = document.getElementById('blGrid');
  if (posts.length === 0) {
    gridEl.innerHTML = '<div class="no-r">No articles found. <button style="background:none;border:none;color:var(--rust);cursor:pointer;font-size:15px;text-decoration:underline" onclick="renderBlog(\'all\',\'\')">Clear filter</button></div>';
    return;
  }
  gridEl.innerHTML = posts.map(function(p) {
    return '<div class="bl-card" onclick="openArticle(\'' + p.id + '\')">' +
      '<div class="bl-ci" style="background:' + p.bg + '">' + p.emoji +
      '<span class="bl-cbadge" style="background:' + p.cc + ';color:' + p.ct + '">' + p.cat + '</span></div>' +
      '<div class="bl-cbody"><h3 class="bl-ct">' + p.title + '</h3><p class="bl-ce">' + p.excerpt + '</p>' +
      '<div class="bl-cm"><span>' + p.date + ' &middot; ' + p.rt + ' read</span><span class="bl-cr">Read &rarr;</span></div></div></div>';
  }).join('');
}

function openArticle(id) {
  var post = blogPosts.find(function(p) { return p.id === id; });
  if (!post) return;
  document.getElementById('blogHub').style.display = 'none';
  var dv = document.getElementById('blogDetail');
  dv.style.display = 'block';
  var related = blogPosts.filter(function(p) { return p.id !== id; }).slice(0, 3);
  var relHtml = related.map(function(r) {
    return '<div class="rp" onclick="openArticle(\'' + r.id + '\')"><div class="rpi" style="background:' + r.bg + '">' + r.emoji + '</div><div><div class="rpt">' + r.title + '</div><div class="rpd">' + r.date + '</div></div></div>';
  }).join('');
  dv.innerHTML =
    '<button class="art-back" onclick="renderBlog(currentBlogCat,\'\')">&#8592; Back to Blog</button>' +
    '<div class="art-lay"><div class="art-main">' +
    '<div class="art-hi" style="background:' + post.bg + '">' + post.emoji + '</div>' +
    '<div class="art-c">' + post.cat + '</div>' +
    '<h1 class="art-t">' + post.title + '</h1>' +
    '<div class="art-m"><span>by ' + post.author + '</span><span>' + post.date + '</span><span>' + post.rt + ' read</span></div>' +
    '<div class="art-body">' + post.body + '</div></div>' +
    '<aside class="art-side">' +
    '<div class="scta"><h4>Get a free quote</h4><p>Free site visit within 24 hours. No commitment required.</p><button class="scta-btn" onclick="goPage(\'contact\')">Book Free Site Visit</button></div>' +
    '<div class="scard"><div class="scard-t">Related Articles</div>' + relHtml + '</div>' +
    '</aside></div>';
  window.scrollTo({ top: 0, behavior: 'smooth' });
}
